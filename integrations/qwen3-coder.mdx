    const authenticateToken = (req, res, next) => {
      const authHeader = req.headers['authorization'];
      const token = authHeader && authHeader.split(' ')[1];
      
      if (!token) {
        return res.status(401).json({ error: 'Access token required' });
      }
      
      jwt.verify(token, process.env.JWT_SECRET, (err, user) => {
        if (err) {

      });
    };

## CI/CD Integration



<CodeGroup>
```yaml GitHub Actions
name: AI Code Review
on: [pull_request]

jobs:
  ai-review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Setup Qwen CLI
        run: npm install -g @qwen-code/qwen-code
      
      - name: Code Review
        run: |
          qwen --model qwen/qwen-3-coder \
               review --format github-comment \
               --files "$(git diff --name-only origin/main)"
        env:
          OPENROUTER_API_KEY: ${{ secrets.OPENROUTER_API_KEY }}
```

```yaml GitLab CI
stages:
  - code-quality

ai-review:
  stage: code-quality
  script:
    - npm install -g @qwen-code/qwen-code
    - qwen --model qwen/qwen-3-coder review --gitlab-format ./src
  only:
    - merge_requests
```
</CodeGroup>

## Best Practices

<AccordionGroup>
  <Accordion title="Code Generation">
    **Do:**
    - Provide clear, specific requirements
    - Include context about your tech stack
    - Specify coding standards and preferences
    - Ask for comments and documentation
    
    **Don't:**
    - Generate code without understanding it
    - Skip testing generated code
    - Ignore security considerations
    - Use without reviewing output
  </Accordion>
  
  <Accordion title="Debugging">
    **Effective Debugging:**
    - Include error messages and stack traces
    - Provide relevant code context
    - Specify your environment (OS, versions)
    - Ask for explanation of the root cause
    
    **Example:**
    ```bash
    qwen debug --error "TypeError: user.map is not a function" \
               --context ./UserList.tsx \
               --environment "React 18, TypeScript 4.8"
    ```
  </Accordion>
  
  <Accordion title="Code Review">
    **Review Focus Areas:**
    - Security vulnerabilities
    - Performance bottlenecks  
    - Code maintainability
    - Best practices adherence
    - Documentation quality
    
    ```bash
    qwen review --focus security,performance,maintainability ./src
    ```
  </Accordion>
</AccordionGroup>
